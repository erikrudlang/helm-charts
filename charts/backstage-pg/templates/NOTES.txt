Backstage has been installed!

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "backstage-pg.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "backstage-pg.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "backstage-pg.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "backstage-pg.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:7007 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 7007:$CONTAINER_PORT
{{- end }}

2. PostgreSQL Database Status:
   
   Check the CloudNativePG cluster status:
   
   kubectl get cluster {{ .Values.postgresql.cluster.name }} -n {{ .Release.Namespace }}
   
   View database pods:
   
   kubectl get pods -l cnpg.io/cluster={{ .Values.postgresql.cluster.name }} -n {{ .Release.Namespace }}
   
   Database credentials are stored in secret:
   
   kubectl get secret {{ .Values.postgresql.cluster.name }}-app -n {{ .Release.Namespace }}

   kubectl rollout restart deployment {{ include "backstage-pg.fullname" . }} -n {{ .Release.Namespace }}
   
   Connect to PostgreSQL:
   kubectl exec -it {{ .Values.postgresql.cluster.name }}-1 -n {{ .Release.Namespace }} -- psql -U {{ .Values.postgresql.auth.username }} -d {{ .Values.postgresql.auth.database }}
